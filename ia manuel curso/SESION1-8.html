1. Conceptos Básicos de Archivos

Los archivos son contenedores de información que se almacenan en el sistema de archivos. En Python, trabajamos con archivos utilizando objetos de archivo.

1.1 Tipos de Archivos

  • Archivos de texto: Contienen texto plano y son legibles por humanos (ej. .txt, .py, .html)
  • Archivos binarios: Contienen datos en formato binario (ej. .jpg, .exe, .pdf)
  • Archivos estructurados: Siguen un formato específico (ej. .csv, .json, .xml)

1.2 Rutas de Archivos

Las rutas pueden ser absolutas o relativas. Es recomendable usar rutas relativas cuando sea posible para mayor portabilidad.

Ejemplo de rutas

# Ruta absoluta (completa)
ruta_absoluta = "C:/Usuarios/MiUsuario/Documentos/archivo.txt"

# Ruta relativa (relativa al directorio actual)
ruta_relativa = "datos/archivo.txt"

# Usar os.path para manejo de rutas multiplataforma
import os
ruta = os.path.join('carpeta', 'subcarpeta', 'archivo.txt')  # 'carpeta/subcarpeta/archivo.txt'

2. Trabajando con Archivos de Texto

2.1 Abrir y Cerrar Archivos

La forma recomendada de trabajar con archivos es usando el contexto with, que se asegura de que el archivo se cierre correctamente.

Modos de apertura de archivos

# Modos básicos
with open('archivo.txt', 'r') as archivo:  # Lectura (por defecto)
    contenido = archivo.read()

with open('archivo.txt', 'w') as archivo:  # Escritura (sobrescribe)
    archivo.write("Nuevo contenido")

with open('archivo.txt', 'a') as archivo:  # Anexar al final
    archivo.write("\nMás contenido")

with open('archivo.txt', 'r+') as archivo:  # Lectura y escritura
    contenido = archivo.read()
    archivo.write("\nActualización")

# Modo binario (para archivos no de texto)
with open('imagen.jpg', 'rb') as archivo_binario:
    datos = archivo_binario.read()

2.2 Leyendo Archivos

Existen varios métodos para leer archivos según las necesidades:

Métodos de lectura

# Leer todo el contenido como una cadena
with open('archivo.txt', 'r') as archivo:
    contenido = archivo.read()

# Leer una línea a la vez
with open('archivo.txt', 'r') as archivo:
    for linea in archivo:
        print(linea.strip())  # strip() elimina saltos de línea

# Leer todas las líneas en una lista
with open('archivo.txt', 'r') as archivo:
    lineas = archivo.readlines()

# Leer una cantidad específica de caracteres
with open('archivo.txt', 'r') as archivo:
    primeros_100 = archivo.read(100)  # Lee los primeros 100 caracteres

2.3 Escribiendo en Archivos

Para escribir en archivos, podemos usar write() o writelines().

Métodos de escritura

# Escribir una cadena
with open('salida.txt', 'w') as archivo:
    archivo.write("Línea 1\n")
    archivo.write("Línea 2\n")

# Escribir múltiples líneas desde una lista
lineas = ["Línea 1\n", "Línea 2\n", "Línea 3\n"]
with open('salida.txt', 'w') as archivo:
    archivo.writelines(lineas)

# Usar print para escribir en archivos (útil para formateo)
with open('salida.txt', 'w') as archivo:
    print("Línea 1", file=archivo)
    print("Línea 2", file=archivo)
    print(f"La suma de 2 + 2 es {2+2}", file=archivo)

3. Trabajando con Archivos CSV

Los archivos CSV (Comma-Separated Values) son comunes para almacenar datos tabulares.

Leyendo archivos CSV

import csv

# Leer un archivo CSV
with open('datos.csv', 'r', newline='', encoding='utf-8') as archivo:
    lector = csv.reader(archivo)
    encabezados = next(lector)  # Leer la primera fila (encabezados)
    for fila in lector:
        print(f"Nombre: {fila[0]}, Edad: {fila[1]}")

# Leer como diccionario (usando encabezados como claves)

    lector = csv.DictReader(archivo)
    for fila in lector:
        print(f"{fila['Nombre']} tiene {fila['Edad']} años")

Escribiendo archivos CSV

import csv

# Datos a guardar
datos = [
    ['Nombre', 'Edad', 'Ciudad'],
    ['Juan', '25', 'Madrid'],
    ['Ana', '30', 'Barcelona'],

with open('personas_dict.csv', 'w', newline='', encoding='utf-8') as archivo:
    campos = ['Nombre', 'Edad', 'Ciudad']
    escritor = csv.DictWriter(archivo, fieldnames=campos)
    escritor.writeheader()
    escritor.writerows(personas)

4. Trabajando con Archivos JSON

JSON (JavaScript Object Notation) es un formato ligero de intercambio de datos.

Leer y escribir archivos JSON

import json

# Datos de ejemplo
datos = {
    "nombre": "Juan Pérez",
    "edad": 30,
    "ciudad": "Madrid",
    "hobbies": ["leer", "programar", "viajar"],
    "activo": True
}

# Escribir en un archivo JSON
with open('usuario.json', 'w', encoding='utf-8') as archivo:
    json.dump(datos, archivo, indent=4, ensure_ascii=False)

# Leer de un archivo JSON
with open('usuario.json', 'r', encoding='utf-8') as archivo:
    datos_cargados = json.load(archivo)
    print(datos_cargados['nombre'])  # Imprime: Juan Pérez

El módulo os proporciona funciones para interactuar con el sistema operativo.

Operaciones comunes con archivos

from pathlib import Path  # Alternativa moderna a os.path

# Obtener el directorio actual
directorio_actual = os.getcwd()

# Listar archivos en un directorio
archivos = os.listdir('.')

if os.path.exists('archivo.txt'):
    print("El archivo existe")

# Crear un directorio
os.makedirs('nuevo_directorio', exist_ok=True)

# Renombrar un archivo
os.rename('viejo.txt', 'nuevo.txt')

# Eliminar un archivo
os.remove('archivo_a_eliminar.txt')

# Eliminar un directorio vacío
os.rmdir('directorio_vacio')

# Eliminar un directorio y su contenido (¡cuidado!)
shutil.rmtree('directorio_con_contenido')

# Usando pathlib (más moderno)
ruta = Path('carpeta') / 'subcarpeta' / 'archivo.txt'
ruta.parent.mkdir(parents=True, exist_ok=True)  # Crear directorios padres si no existen
ruta.write_text('Contenido del archivo')  # Escribir en el archivo

6. Manejo de Excepciones con Archivos

Es importante manejar posibles errores al trabajar con archivos.

Manejo de errores

try:
    with open('archivo_inexistente.txt', 'r') as archivo:
        contenido = archivo.read()
except FileNotFoundError:
    print("El archivo no existe")
except PermissionError:
    print("No tienes permiso para leer el archivo")
except IOError as e:
    print(f"Error de E/S: {e}")
except Exception as e:
    print(f"Error inesperado: {e}")
else:
    print("Archivo leído correctamente")
finally:
    print("Este bloque siempre se ejecuta")

Ejercicio 1: Contador de Palabras

Crea un programa que lea un archivo de texto y cuente la frecuencia de cada palabra, guardando los resultados en un archivo CSV.

import re
from collections import Counter
import csv

def limpiar_palabra(palabra):
    """Elimina caracteres no alfabéticos y convierte a minúsculas"""
    return re.sub(r'[^\w]', '', palabra).lower()

def contar_palabras(archivo_entrada, archivo_salida):
    try:
        # Leer el archivo de entrada
        with open(archivo_entrada, 'r', encoding='utf-8') as archivo:
            texto = archivo.read()
        
        # Dividir en palabras y limpiar
        palabras = texto.split()
        palabras_limpias = [limpiar_palabra(p) for p in palabras if limpiar_palabra(p)]
        
        # Contar frecuencias
        contador = Counter(palabras_limpias)
        
        # Ordenar por frecuencia (de mayor a menor)
        palabras_comunes = contador.most_common()
        
        # Guardar resultados en CSV
        with open(archivo_salida, 'w', newline='', encoding='utf-8') as archivo_csv:
            escritor = csv.writer(archivo_csv)
            escritor.writerow(['Palabra', 'Frecuencia'])
            escritor.writerows(palabras_comunes)
        
        print(f"Análisis completado. Resultados guardados en {archivo_salida}")
        
    except FileNotFoundError:
        print(f"Error: No se pudo encontrar el archivo {archivo_entrada}")
    except Exception as e:
        print(f"Ocurrió un error: {e}")

if __name__ == "__main__":
    # Ejemplo de uso
    archivo_entrada = input("Ruta del archivo de texto: ")
    archivo_salida = input("Ruta del archivo de salida CSV: ")
    contar_palabras(archivo_entrada, archivo_salida)

Ejercicio 2: Gestor de Tareas con Archivos

Amplía el gestor de tareas de la sesión anterior para que guarde las tareas en un archivo JSON y las cargue al iniciar.

import json
import os

ARCHIVO_TAREAS = 'tareas.json'

def cargar_tareas():
    """Carga las tareas desde el archivo JSON."""
    if os.path.exists(ARCHIVO_TAREAS):
        try:
            with open(ARCHIVO_TAREAS, 'r', encoding='utf-8') as archivo:
                return json.load(archivo)
        except (json.JSONDecodeError, IOError) as e:
            print(f"Error al cargar las tareas: {e}")
            print("Se creará una nueva lista de tareas.")
    return []

def guardar_tareas(tareas):
    """Guarda las tareas en el archivo JSON."""
    try:
        with open(ARCHIVO_TAREAS, 'w', encoding='utf-8') as archivo:
            json.dump(tareas, archivo, indent=4, ensure_ascii=False)
        return True
    except IOError as e:
        print(f"Error al guardar las tareas: {e}")
        return False

def mostrar_tareas(tareas):
    """Muestra las tareas numeradas."""
    if not tareas:
        print("No hay tareas pendientes.")
        return
    
    print("\n--- Tareas Pendientes ---")
    for i, tarea in enumerate(tareas, 1):
        estado = "[✓]" if tarea.get('completada', False) else "[ ]"
        print(f"{i}. {estado} {tarea['descripcion']}")

def agregar_tarea(tareas, descripcion):
    """Agrega una nueva tarea."""
    tarea = {
        'descripcion': descripcion,
        'completada': False,
        'fecha_creacion': datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    }
    tareas.append(tarea)
    if guardar_tareas(tareas):
        print("\n¡Tarea agregada con éxito!")
    return tareas

def marcar_completada(tareas, indice):
    """Marca una tarea como completada."""
    if 1 <= indice <= len(tareas):
        tareas[indice-1]['completada'] = True
        tareas[indice-1]['fecha_completada'] = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        if guardar_tareas(tareas):
            print("\n¡Tarea marcada como completada!")
    else:
        print("\n¡Número de tarea inválido!")
    return tareas

def eliminar_tarea(tareas, indice):
    """Elimina una tarea."""
    if 1 <= indice <= len(tareas):
        tarea_eliminada = tareas.pop(indice-1)
        if guardar_tareas(tareas):
            print(f"\nTarea eliminada: {tarea_eliminada['descripcion']}")
    else:
        print("\n¡Número de tarea inválido!")
    return tareas

def mostrar_menu():
    """Muestra el menú de opciones."""
    print("\n--- Gestor de Tareas ---")
    print("1. Ver tareas")
    print("2. Agregar tarea")
    print("3. Marcar tarea como completada")
    print("4. Eliminar tarea")
    print("5. Salir")
    return input("Selecciona una opción: ")

def main():
    """Función principal del programa."""
    tareas = cargar_tareas()
    
    while True:
        opcion = mostrar_menu()
        
        if opcion == '1':
            mostrar_tareas(tareas)
            
        elif opcion == '2':
            descripcion = input("\nDescripción de la tarea: ").strip()
            if descripcion:
                tareas = agregar_tarea(tareas, descripcion)
            else:
                print("La descripción no puede estar vacía.")
                
        elif opcion == '3':
            mostrar_tareas(tareas)
            if tareas:
                try:
                    num = int(input("\nNúmero de tarea a marcar como completada: "))
                    tareas = marcar_completada(tareas, num)
                except ValueError:
                    print("Por favor, ingresa un número válido.")
                    
        elif opcion == '4':
            mostrar_tareas(tareas)
            if tareas:
                try:
                    num = int(input("\nNúmero de tarea a eliminar: "))
                    confirmacion = input("¿Estás seguro? (s/n): ")
                    if confirmacion.lower() == 's':
                        tareas = eliminar_tarea(tareas, num)
                except ValueError:
                    print("Por favor, ingresa un número válido.")
                    
        elif opcion == '5':
            print("\n¡Hasta luego!")
            break
            
        else:
            print("\nOpción no válida. Por favor, intenta de nuevo.")

if __name__ == "__main__":
    from datetime import datetime
    main()

Ejercicio 3: Organizador de Archivos

Crea un script que organice los archivos de un directorio en subdirectorios según su extensión.

import os
import shutil
from pathlib import Path

def crear_directorios(directorio_base, extensiones):
    """Crea los directorios necesarios para organizar los archivos."""
    for extension in extensiones:
        ruta_directorio = directorio_base / extension.upper()
        ruta_directorio.mkdir(exist_ok=True)

def obtener_extension(archivo):
    """Obtiene la extensión de un archivo, devuelve 'SIN_EXTENSION' si no tiene."""
    extension = os.path.splitext(archivo)[1].lower()
    return extension[1:] if extension else 'SIN_EXTENSION'

def organizar_archivos(directorio):
    """Organiza los archivos del directorio en subdirectorios por extensión."""
    try:
        # Convertir a objeto Path si es una cadena
        directorio = Path(directorio) if isinstance(directorio, str) else directorio
        
        if not directorio.exists() or not directorio.is_dir():
            print(f"Error: '{directorio}' no es un directorio válido.")
            return
        
        # Obtener todos los archivos (excluyendo directorios)
        archivos = [f for f in directorio.iterdir() if f.is_file()]
        
        if not archivos:
            print(f"No se encontraron archivos en '{directorio}'.")
            return
        
        # Obtener todas las extensiones únicas
        extensiones = {obtener_extension(archivo.name) for archivo in archivos}
        
        # Crear directorios para cada extensión
        crear_directorios(directorio, extensiones)
        
        # Mover archivos a sus respectivos directorios
        for archivo in archivos:
            extension = obtener_extension(archivo.name)
            directorio_destino = directorio / extension.upper()
            
            try:
                # Verificar si el archivo ya existe en el destino
                destino = directorio_destino / archivo.name
                if destino.exists():
                    # Si el archivo ya existe, agregar un sufijo numérico
                    contador = 1
                    while destino.exists():
                        nombre_base = f"{archivo.stem}_{contador}{archivo.suffix}"
                        destino = directorio_destino / nombre_base
                        contador += 1
                
                shutil.move(str(archivo), str(destino))
                print(f"Movido: {archivo.name} -> {directorio_destino.name}/")
                
            except Exception as e:
                print(f"Error al mover {archivo.name}: {e}")
        
        print("\n¡Organización completada con éxito!")
        
    except Exception as e:
        print(f"Ocurrió un error: {e}")

def main():
    """Función principal del programa."""
    print("--- Organizador de Archivos ---\n")
    
    # Obtener el directorio actual si no se especifica uno
    directorio_actual = os.getcwd()
    ruta = input(f"Ingresa la ruta del directorio a organizar (Enter para usar '{directorio_actual}'): ")
    
    # Usar el directorio actual si no se proporciona una ruta
    ruta = ruta.strip() if ruta.strip() else directorio_actual
    
    # Mostrar advertencia
    print("\n¡ADVERTENCIA: Este programa moverá los archivos a subdirectorios según su extensión.")
    print("Asegúrate de hacer una copia de seguridad de tus archivos importantes.")
    confirmacion = input("\n¿Deseas continuar? (s/n): ")
    
    if confirmacion.lower() == 's':
        organizar_archivos(ruta)
    else:
        print("Operación cancelada.")

if __name__ == "__main__":
    main()
