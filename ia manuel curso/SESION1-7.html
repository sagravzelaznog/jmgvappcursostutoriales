1. Diccionarios en Python

Los diccionarios son estructuras de datos que almacenan pares clave-valor. Son mutables y no permiten claves duplicadas.

Creación de diccionarios

# Diccionario vacío
mi_diccionario = {}
otro_diccionario = dict()

# Diccionario con valores iniciales
estudiante = {
    'nombre': 'María',
    'edad': 22,
    'cursos': ['Matemáticas', 'Física', 'Programación'],
    'activo': True
}

1.1 Acceso y modificación

# Acceso a valores
print(estudiante['nombre'])  # 'María'
print(estudiante.get('edad'))  # 22
print(estudiante.get('direccion', 'No especificada'))  # Valor por defecto

# Modificación de valores
estudiante['edad'] = 23
estudiante['direccion'] = 'Calle Falsa 123'  # Añadir nuevo par clave-valor

1.2 Métodos útiles de diccionarios

Método Descripción Ejemplo
keys() Devuelve las claves del diccionario estudiante.keys()
values() Devuelve los valores del diccionario estudiante.values()
items() Devuelve pares (clave, valor) estudiante.items()
update() Actualiza el diccionario con pares clave-valor de otro estudiante.update({'telefono': '123-456-789'})
pop() Elimina y devuelve el valor de una clave edad = estudiante.pop('edad')

2. Conjuntos en Python

Los conjuntos son colecciones desordenadas de elementos únicos. Son útiles para operaciones de teoría de conjuntos.

Creación de conjuntos

# Conjunto vacío
conjunto_vacio = set()  # ¡No usar {} que crea un diccionario!

# Conjunto con valores
frutas = {'manzana', 'banana', 'cereza'}
numeros = {1, 2, 3, 4, 5}

# Convertir lista a conjunto (elimina duplicados)
numeros_repetidos = [1, 2, 2, 3, 3, 3, 4]
numeros_unicos = set(numeros_repetidos)  # {1, 2, 3, 4}

2.1 Operaciones con conjuntos

# Unión
A = {1, 2, 3}
B = {3, 4, 5}
print(A | B)  # {1, 2, 3, 4, 5}

# Intersección
print(A & B)  # {3}

# Diferencia
print(A - B)  # {1, 2}

# Diferencia simétrica
print(A ^ B)  # {1, 2, 4, 5}

2.2 Métodos útiles de conjuntos

Método Descripción Ejemplo
add() Añade un elemento al conjunto frutas.add('naranja')
remove() Elimina un elemento (genera error si no existe) frutas.remove('banana')
discard() Elimina un elemento si existe frutas.discard('kiwi')
update() Añade múltiples elementos frutas.update(['mango', 'piña'])
issubset() Comprueba si es subconjunto {1, 2}.issubset({1, 2, 3}) # True

Ejercicio 1: Gestor de Contactos

Crea un programa que permita gestionar una agenda de contactos usando diccionarios. Debe permitir agregar, buscar, actualizar y eliminar contactos.

def mostrar_menu():
    print("\n--- Gestor de Contactos ---")
    print("1. Ver todos los contactos")
    print("2. Buscar contacto")
    print("3. Agregar contacto")
    print("4. Actualizar contacto")
    print("5. Eliminar contacto")
    print("6. Salir")
    return input("Selecciona una opción: ")

def main():
    # Diccionario para almacenar los contactos
    # Estructura: {'email': {'nombre': '...', 'telefono': '...'}}
    contactos = {}
    
    while True:
        opcion = mostrar_menu()
        
        if opcion == '1':
            # Ver todos los contactos
            if not contactos:
                print("\nNo hay contactos guardados.")
            else:
                print("\n--- Contactos ---")
                for email, datos in contactos.items():
                    print(f"\nEmail: {email}")
                    print(f"Nombre: {datos['nombre']}")
                    print(f"Teléfono: {datos.get('telefono', 'No especificado')}")
                    
        elif opcion == '2':
            # Buscar contacto
            email = input("\nIngresa el email del contacto: ").strip().lower()
            contacto = contactos.get(email)
            if contacto:
                print(f"\nNombre: {contacto['nombre']}")
                print(f"Email: {email}")
                print(f"Teléfono: {contacto.get('telefono', 'No especificado')}")
            else:
                print("\nContacto no encontrado.")
                
        elif opcion == '3':
            # Agregar contacto
            print("\n--- Agregar Contacto ---")
            email = input("Email: ").strip().lower()
            
            if email in contactos:
                print("\n¡Ya existe un contacto con ese email!")
                continue
                
            nombre = input("Nombre: ").strip()
            telefono = input("Teléfono (opcional): ").strip()
            
            contactos[email] = {'nombre': nombre}
            if telefono:
                contactos[email]['telefono'] = telefono
                
            print("\n¡Contacto agregado con éxito!")
            
        elif opcion == '4':
            # Actualizar contacto
            email = input("\nIngresa el email del contacto a actualizar: ").strip().lower()
            
            if email not in contactos:
                print("\nContacto no encontrado.")
                continue
                
            print("\nDeja en blanco los campos que no quieras modificar.")
            nuevo_nombre = input(f"Nuevo nombre ({contactos[email]['nombre']}): ").strip()
            nuevo_telefono = input(f"Nuevo teléfono ({contactos[email].get('telefono', 'No especificado')}): ").strip()
            
            if nuevo_nombre:
                contactos[email]['nombre'] = nuevo_nombre
            if nuevo_telefono:
                contactos[email]['telefono'] = nuevo_telefono
                
            print("\n¡Contacto actualizado con éxito!")
            
        elif opcion == '5':
            # Eliminar contacto
            email = input("\nIngresa el email del contacto a eliminar: ").strip().lower()
            
            if email in contactos:
                confirmacion = input(f"¿Estás seguro de eliminar a {contactos[email]['nombre']}? (s/n): ")
                if confirmacion.lower() == 's':
                    del contactos[email]
                    print("\n¡Contacto eliminado con éxito!")
            else:
                print("\nContacto no encontrado.")
                
        elif opcion == '6':
            # Salir
            print("\n¡Hasta luego!")
            break
            
        else:
            print("\nOpción no válida. Por favor, intenta de nuevo.")

if __name__ == "__main__":
    main()

Ejercicio 2: Análisis de Texto

Escribe un programa que analice un texto y muestre las palabras más comunes y las letras únicas que contiene.

import string

def limpiar_palabra(palabra):
    """Elimina signos de puntuación y convierte a minúsculas"""
    return palabra.strip(string.punctuation).lower()

def analizar_texto():
    print("--- Analizador de Texto ---")
    print("Ingresa un texto (escribe 'fin' en una línea nueva para terminar):")
    
    # Leer múltiples líneas de entrada
    lineas = []
    while True:
        linea = input()
        if linea.lower() == 'fin':
            break
        lineas.append(linea)
    
    if not lineas:
        print("No se ingresó ningún texto.")
        return
    
    texto = ' '.join(lineas)
    
    # 1. Contar palabras
    palabras = [limpiar_palabra(palabra) for palabra in texto.split()]
    palabras = [p for p in palabras if p]  # Eliminar cadenas vacías
    
    if not palabras:
        print("No se encontraron palabras válidas en el texto.")
        return
    
    # Contar frecuencia de palabras
    frecuencia_palabras = {}
    for palabra in palabras:
        if palabra in frecuencia_palabras:
            frecuencia_palabras[palabra] += 1
        else:
            frecuencia_palabras[palabra] = 1
    
    # Ordenar palabras por frecuencia (de mayor a menor)
    palabras_comunes = sorted(frecuencia_palabras.items(), key=lambda x: x[1], reverse=True)
    
    # 2. Encontrar letras únicas
    letras_unicas = set()
    for letra in texto.lower():
        if letra.isalpha():
            letras_unicas.add(letra)
    
    # Mostrar resultados
    print("\n--- Resultados del Análisis ---")
    
    print("\nPalabras más comunes:")
    for palabra, frecuencia in palabras_comunes[:5]:  # Mostrar solo las 5 más comunes
        print(f"{palabra}: {frecuencia} veces")
    
    print(f"\nTotal de palabras únicas: {len(frecuencia_palabras)}")
    print(f"Total de palabras: {len(palabras)}")
    
    print("\nLetras únicas encontradas:", ''.join(sorted(letras_unicas)))
    print(f"Total de letras únicas: {len(letras_unicas)}")

if __name__ == "__main__":
    analizar_texto()

Ejercicio 3: Sistema de Inventario

Crea un sistema de inventario que permita gestionar productos, sus cantidades y precios utilizando diccionarios anidados.

def mostrar_menu():
    print("\n--- Sistema de Inventario ---")
    print("1. Ver todos los productos")
    print("2. Buscar producto por ID")
    print("3. Agregar nuevo producto")
    print("4. Actualizar stock")
    print("5. Realizar venta")
    print("6. Generar informe de inventario")
    print("7. Salir")
    return input("Selecciona una opción: ")

def main():
    # Estructura: {id: {'nombre': str, 'precio': float, 'stock': int, 'categoria': str}}
    inventario = {
        'P001': {'nombre': 'Laptop', 'precio': 15000.0, 'stock': 10, 'categoria': 'Electrónicos'},
        'P002': {'nombre': 'Mouse', 'precio': 250.0, 'stock': 50, 'categoria': 'Accesorios'},
        'P003': {'nombre': 'Teclado', 'precio': 350.0, 'stock': 30, 'categoria': 'Accesorios'},
    }
    
    while True:
        opcion = mostrar_menu()
        
        if opcion == '1':
            # Ver todos los productos
            if not inventario:
                print("\nNo hay productos en el inventario.")
                continue
                
            print("\n--- Productos en Inventario ---")
            print("ID\tNombre\t\tPrecio\tStock\tCategoría")
            print("-" * 60)
            for id_producto, producto in inventario.items():
                print(f"{id_producto}\t{producto['nombre']}\t${producto['precio']:.2f}\t"
                      f"{producto['stock']}\t{producto['categoria']}")
        
        elif opcion == '2':
            # Buscar producto por ID
            id_producto = input("\nIngresa el ID del producto: ").upper()
            producto = inventario.get(id_producto)
            
            if producto:
                print("\n--- Detalles del Producto ---")
                print(f"ID: {id_producto}")
                print(f"Nombre: {producto['nombre']}")
                print(f"Precio: ${producto['precio']:.2f}")
                print(f"Stock disponible: {producto['stock']}")
                print(f"Categoría: {producto['categoria']}")
            else:
                print("\nProducto no encontrado.")
        
        elif opcion == '3':
            # Agregar nuevo producto
            print("\n--- Agregar Nuevo Producto ---")
            
            # Generar ID automático (P + número secuencial)
            if not inventario:
                nuevo_id = 'P001'
            else:
                # Encontrar el máximo ID existente y sumar 1
                max_id = max(int(k[1:]) for k in inventario.keys())
                nuevo_id = f"P{max_id + 1:03d}"
            
            print(f"ID generado: {nuevo_id}")
            
            # Validar nombre
            while True:
                nombre = input("Nombre del producto: ").strip()
                if nombre:
                    break
                print("El nombre no puede estar vacío.")
            
            # Validar precio
            while True:
                try:
                    precio = float(input("Precio unitario: $"))
                    if precio > 0:
                        break
                    print("El precio debe ser mayor a 0.")
                except ValueError:
                    print("Por favor, ingresa un número válido.")
            
            # Validar stock
            while True:
                try:
                    stock = int(input("Cantidad en stock: "))
                    if stock >= 0:
                        break
                    print("El stock no puede ser negativo.")
                except ValueError:
                    print("Por favor, ingresa un número entero.")
            
            # Categoría
            categoria = input("Categoría: ").strip() or "Sin categoría"
            
            # Agregar al inventario
            inventario[nuevo_id] = {
                'nombre': nombre,
                'precio': precio,
                'stock': stock,
                'categoria': categoria
            }
            
            print(f"\n¡Producto '{nombre}' agregado con éxito!")
        
        elif opcion == '4':
            # Actualizar stock
            id_producto = input("\nIngresa el ID del producto a actualizar: ").upper()
            
            if id_producto not in inventario:
                print("Producto no encontrado.")
                continue
                
            producto = inventario[id_producto]
            print(f"\nProducto: {producto['nombre']}")
            print(f"Stock actual: {producto['stock']}")
            
            try:
                cambio = int(input("Ingresa la cantidad a sumar (o restar con -): "))
                nuevo_stock = producto['stock'] + cambio
                
                if nuevo_stock < 0:
                    print("No puedes tener un stock negativo.")
                else:
                    producto['stock'] = nuevo_stock
                    print(f"Stock actualizado: {nuevo_stock}")
            except ValueError:
                print("Por favor, ingresa un número entero.")
        
        elif opcion == '5':
            # Realizar venta
            print("\n--- Realizar Venta ---")
            
            if not inventario:
                print("No hay productos en el inventario.")
                continue
                
            carrito = {}
            total_venta = 0.0
            
            while True:
                print("\n--- Carrito de Compras ---")
                if carrito:
                    print("Productos en el carrito:")
                    for id_producto, cantidad in carrito.items():
                        producto = inventario[id_producto]
                        subtotal = producto['precio'] * cantidad
                        print(f"- {producto['nombre']}: {cantidad} x ${producto['precio']:.2f} = ${subtotal:.2f}")
                    print(f"Total: ${total_venta:.2f}")
                else:
                    print("El carrito está vacío.")
                
                print("\n1. Agregar producto")
                print("2. Finalizar venta")
                print("3. Cancelar")
                opcion_venta = input("Selecciona una opción: ")
                
                if opcion_venta == '1':
                    id_producto = input("\nIngresa el ID del producto: ").upper()
                    
                    if id_producto not in inventario:
                        print("Producto no encontrado.")
                        continue
                        
                    producto = inventario[id_producto]
                    print(f"Producto: {producto['nombre']}")
                    print(f"Precio unitario: ${producto['precio']:.2f}")
                    print(f"Stock disponible: {producto['stock']}")
                    
                    try:
                        cantidad = int(input("Cantidad: "))
                        if cantidad <= 0:
                            print("La cantidad debe ser mayor a 0.")
                            continue
                            
                        if cantidad > producto['stock']:
                            print(f"No hay suficiente stock. Solo hay {producto['stock']} disponibles.")
                            continue
                            
                        # Agregar al carrito o actualizar cantidad
                        if id_producto in carrito:
                            carrito[id_producto] += cantidad
                        else:
                            carrito[id_producto] = cantidad
                            
                        # Actualizar total
                        total_venta += producto['precio'] * cantidad
                        print("\n¡Producto agregado al carrito!")
                        
                    except ValueError:
                        print("Por favor, ingresa un número válido.")
                
                elif opcion_venta == '2':
                    if not carrito:
                        print("No hay productos en el carrito.")
                        continue
                        
                    # Mostrar resumen de la venta
                    print("\n--- Resumen de la Venta ---")
                    for id_producto, cantidad in carrito.items():
                        producto = inventario[id_producto]
                        subtotal = producto['precio'] * cantidad
                        print(f"- {producto['nombre']}: {cantidad} x ${producto['precio']:.2f} = ${subtotal:.2f}")
                    
                    print(f"\nTotal a pagar: ${total_venta:.2f}")
                    
                    # Confirmar venta
                    confirmacion = input("\n¿Confirmar venta? (s/n): ").lower()
                    if confirmacion == 's':
                        # Actualizar inventario
                        for id_producto, cantidad in carrito.items():
                            inventario[id_producto]['stock'] -= cantidad
                        
                        print("\n¡Venta realizada con éxito!")
                        print("¡Gracias por su compra!")
                        carrito = {}
                        total_venta = 0.0
                    
                    break
                
                elif opcion_venta == '3':
                    print("\nVenta cancelada.")
                    break
                
                else:
                    print("\nOpción no válida. Intenta de nuevo.")
        
        elif opcion == '6':
            # Generar informe
            if not inventario:
                print("\nNo hay productos en el inventario.")
                continue
                
            print("\n--- Informe de Inventario ---")
            print(f"Total de productos: {len(inventario)}")
            
            # Calcular valor total del inventario
            valor_total = sum(p['precio'] * p['stock'] for p in inventario.values())
            print(f"Valor total del inventario: ${valor_total:.2f}")
            
            # Producto con mayor stock
            producto_max_stock = max(inventario.items(), key=lambda x: x[1]['stock'])
            print(f"\nProducto con mayor stock: {producto_max_stock[1]['nombre']} ({producto_max_stock[1]['stock']} unidades)")
            
            # Producto más caro
            producto_mas_caro = max(inventario.items(), key=lambda x: x[1]['precio'])
            print(f"Producto más caro: {producto_mas_caro[1]['nombre']} (${producto_mas_caro[1]['precio']:.2f})")
            
            # Productos con stock bajo (menos de 5 unidades)
            productos_bajo_stock = [(id_p, p) for id_p, p in inventario.items() if p['stock'] < 5]
            if productos_bajo_stock:
                print("\n¡Atención! Los siguientes productos tienen bajo stock:")
                for id_p, p in productos_bajo_stock:
                    print(f"- {p['nombre']} (ID: {id_p}): {p['stock']} unidades")
        
        elif opcion == '7':
            # Salir
            print("\n¡Hasta luego!")
            break
        
        else:
            print("\nOpción no válida. Por favor, intenta de nuevo.")

if __name__ == "__main__":
    main()
